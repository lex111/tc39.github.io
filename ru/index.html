Приватные методы экземпляра класса и аксессоры

  • Январь 2019
  • Тесты
  • Спецификация
Автор: Daniel Ehrenberg | Чемпионы: Daniel Ehrenberg, Kevin Gibbons
Сохранение состояния и поведения приватным в классе позволяет авторам библиотек предоставить понятный и стабильный интерфейс, при этом изменяя свой код незаметно для пользователей.
Показать пример 
class Counter extends HTMLElement {
  #xValue = 0;
  get #x() { return #xValue; }
  set #x(value) {
    this.#xValue = value;
    window.requestAnimationFrame(this.#render.bind(this));
  }
  #clicked() {
    this.#x++;
  }
  constructor() {
    super();
    this.onclick = this.#clicked.bind(this);
  }
  connectedCallback() { this.#render(); }
  #render() {
    this.textContent = this.#x.toString();
  }
}
window.customElements.define('num-counter', Counter);

Статические поля класса и приватные статические методы

  • Январь 2019
  • Тесты
  • Спецификация
Авторы: Daniel Ehrenberg, Kevin Gibbons, Jeff Morrison, Kevin Smith | Чемпионы: Shu-Yu Guo, Daniel Ehrenberg
Предложение по добавлению новой функциональности в JavaScript классах, основыванное на предыдущих предложениях о полях класса и приватных методах.
Показать пример 
class ColorFinder {
  static #red = "#ff0000";
  static #green = "#00ff00";
  static #blue = "#0000ff";

  static colorName(name) {
    switch (name) {
      case "red": return ColorFinder.#red;
      case "blue": return ColorFinder.#blue;
      case "green": return ColorFinder.#green;
      default: throw new RangeError("unknown color");
    }
  }

  // Somehow use colorName
}

await на верхнем уровне модуля

  • Июнь 2019
  • Тесты
  • Спецификация
Авторы и чемпионы: Myles Borins
Top-level await позволяет использовать модули как асинхронные функции. ECMAScript модули (ESM) могут ожидать внешние ресурсы, заставляя другие модули, которые их импортируют, ждать, прежде чем они начнут исполняться.
Показать пример 
// file.html
<script type=module src="a.mjs"></script>
// a.mjs
await import("./b.mjs");
// b.mjs
await import("./a.mjs");

Индексы соответствия регулярных выражений

  • Декабрь 2019
  • Тесты
Авторы и чемпионы: Ron Buckton
Индексы соответствия регулярных выражений в ECMAScript предоставляют дополнительную информацию о начальном и конечном индексах захваченных подстрок относительно начала входной строки.
Показать пример 
const re1 = /a+(?<Z>z)?/;
// indices are relative to start of the input string:
const s1 = "xaaaz";
const m1 = re1.exec(s1);
m1.indices[0][0] === 1;
m1.indices[0][1] === 5;
s1.slice(...m1.indices[0]) === "aaaz";
m1.indices[1][0] === 4;
m1.indices[1][1] === 5;
s1.slice(...m1.indices[1]) === "z";
m1.indices.groups["Z"][0] === 4;
m1.indices.groups["Z"][1] === 5;
s1.slice(...m1.indices.groups["Z"]) === "z";
// capture groups that are not matched return `undefined`:
const m2 = re1.exec("xaaay");
m2.indices[1] === undefined;
m2.indices.groups["Z"] === undefined;

.item()

  • Сентябрь 2020
  • Тесты
  • Спецификация
Авторы и чемпионы: Shu-yu Guo, Tab Atkins
Предложение по добавлению метода .item() во все основные перечисляемые классы (Array, String, TypedArray).
Показать пример 
function item(n) {
	// ToInteger() abstract op
	n = Math.trunc(n) || 0;
	// Allow negative indexing from the end
	if(n < 0) n += this.length;
	// OOB access is guaranteed to return undefined
	if(n < 0 || n >= this.length) return undefined;
	// Otherwise, this is just normal property access
	return this[n];
}
// Other TypedArray constructors omitted for brevity.
for (let C of [Array, String, Uint8Array]) {
    Object.defineProperty(C.prototype, "item",
                          { value: item,
                            writable: true,
                            enumerable: false,
                            configurable: true });
}
